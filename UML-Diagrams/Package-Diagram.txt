@startuml
package "Location" {
    class Province {
        +province_id : Integer
    }

    class City {
        +city_id : Integer
    }

    class Branch {
        +branch_id : Integer
    }
}

package "Scheduling" {
    class Schedule {
        +schedule_id : Integer
    }

    class TimeSlot {
        +time_slot_id : Integer
        +start_time : DateTime
        +end_time : DateTime
        +isReserved : Boolean
    }

    class TimeSlotCatalogue {
        +catalogue_id : Integer
    }
}

package "Users" {
    class Instructor {
        +instructor_id : Integer
        +name : String
        +phone : String
        +specialization : String
        +email : String
        +password : String
    }

    class Client {
        +client_id : Integer
        +age : Integer
        +guardian_id : Null, Integer
        +email : String
        +password : String
    }

    class Administrator {
        +admin_id : Integer
        +email : String
        +password : String
    }
}

package "Offerings" {
    class Offering {
        +offering_id : Integer
        +lesson_type : String
        +mode : String
        +capacity: Integer
    }

    class PublicOffering {
        +public_offering_id : Integer
        +booking_status : String <<enum>>
        +max_clients: Integer
    }
}

' Aggregations
Province o-- City
City o-- Branch
Schedule o-- PublicOffering
TimeSlotCatalogue o-- TimeSlot

' Relationships
Instructor "1" -- "0..*" City : is available in
Instructor "1" -- "0..*" Offering : accepts
Instructor "1" -- "0..*" PublicOffering : creates
Client "1" -- "0..*" PublicOffering : registers
Administrator "1" -- "0..*" Offering : manages
Administrator "1" -- "0..*" Instructor : manages
Administrator "1" -- "0..*" Client : manages
Offering  -- TimeSlotCatalogue
Instructor "1" -- "1" Schedule : has
Client "1" -- "1" Schedule : has
Branch "1" -- "1" Schedule : has

'Inheritance
Offering <|-- PublicOffering : inheritance


@enduml
